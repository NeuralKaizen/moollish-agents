# UI del Agente 1 (Funding Officer) — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Construir la UI web del Agente 1: una pantalla única donde se pega el texto de una convocatoria y se renderiza el `OpportunityAnalysis` completo (semáforo, score explicable, fit por vehículo, evidencia, gaps, partners, riesgos, acciones, borradores), integrada de verdad contra el núcleo `lib/agent/` vía un API route.

**Architecture:** Se monta Next.js (App Router) + Tailwind v4 + shadcn/ui sobre el proyecto TS existente, sin tocar `lib/agent`. Un Route Handler server-side (`app/api/analyze`) invoca `analyzeOpportunity(text, { generate: generateWithOpenRouter })` y devuelve el `OpportunityAnalysis`. El cliente llama a ese route a través de un único punto (`analyzeClient`), con un modo fixture para iterar diseño sin LLM. Los componentes son presentacionales y tipan contra `@/lib/agent/schema`.

**Tech Stack:** Next.js 16 (App Router) · React 19 · TypeScript (strict, ESM) · Tailwind CSS v4 · shadcn/ui · Vitest (solo para la lógica pura de `lib/ui/format.ts` y la validación del fixture) · pnpm.

## Global Constraints

- TypeScript `strict`, ESM (`"type": "module"`), `moduleResolution: "bundler"`, alias `@/*` → `./*`.
- Node v26, pnpm v10. Gestor de paquetes: **pnpm** siempre.
- **NO modificar `lib/agent/`** (núcleo, lo mantiene otra sesión). La UI solo lo importa.
- Los componentes y el route tipan contra el contrato real en `lib/agent/schema.ts`. Campos a respetar exactamente: `deadline = { date: string|null, verified: boolean }` (sin `days_remaining`: se calcula en `format.ts`); `funding_amount = { value, currency, confirmed, estimated_cop, estimated_usd }` (sin `range_*`); `effort`/`risk` y `institutional_fit` son top-level.
- Copy de la UI en **español**.
- Paleta arena + naranja Moollish (tokens en §Task 2). Ancho de lectura máx. ~768px (`max-w-3xl`), layout centrado, orden decision-first.
- Provider del LLM: OpenRouter vía AI SDK (ya implementado en el núcleo). `OPENROUTER_API_KEY` solo server-side.
- Commits frecuentes, uno por task. Mensajes en español, prefijo convencional (`feat:`, `chore:`, `test:`).

```ts
import type { OpportunityAnalysis } from '@/lib/agent/schema'
import { DEFAULT_WEIGHTS } from '@/lib/agent/config'

type Semaforo = OpportunityAnalysis['semaforo']
type Recommendation = OpportunityAnalysis['recommendation']
type Vehicle = OpportunityAnalysis['recommended_vehicle']
type CriterionKey = keyof OpportunityAnalysis['criteria_scores']
type Level = OpportunityAnalysis['effort']

export const SEMAFORO_META: Record<Semaforo, { label: string; color: string }> = {
  verde_alto: { label: 'Verde alto', color: '#3c7d34' },
  verde_condicionado: { label: 'Verde condicionado', color: '#3c7d34' },
  amarillo: { label: 'Amarillo', color: '#9a6b12' },
  naranja: { label: 'Naranja', color: '#c2611c' },
  rojo: { label: 'Rojo', color: '#b23a2e' },
}

export const RECOMMENDATION_LABEL: Record<Recommendation, string> = {
  apply_now: 'Aplicar ya',
  apply_with_partner: 'Aplicar con socio',
  observe: 'Observar',
  request_info: 'Pedir información',
  discard: 'Descartar',
}

export const VEHICLE_LABEL: Record<Vehicle, string> = {
  moollish: 'Moollish',
  moollish_sat2farm: 'Moollish + Sat2Farm',
  foundation_nova: 'Foundation Nova',
  alianza: 'Alianza',
}

export const CRITERION_LABEL: Record<CriterionKey, string> = {
  alineacion_estrategica: 'Alineación estratégica',
  elegibilidad: 'Elegibilidad jurídica/institucional',
  monto_retorno: 'Monto y retorno esperado',
  probabilidad_exito: 'Probabilidad de éxito',
  complejidad_documental: 'Complejidad documental',
  tiempo_disponible: 'Tiempo disponible',
  impacto_estrategico: 'Impacto estratégico',
  riesgo_ejecucion: 'Riesgo de ejecución',
}

export const LEVEL_LABEL: Record<Level, string> = {
  bajo: 'Bajo',
  medio: 'Medio',
  alto: 'Alto',
}

export function criterionWeightPct(key: CriterionKey): number {
  return Math.round(DEFAULT_WEIGHTS[key] * 100)
}

export function daysRemaining(isoDate: string | null, now: Date = new Date()): number | null {
  if (!isoDate) return null
  const target = new Date(isoDate).getTime()
  if (Number.isNaN(target)) return null
  return Math.ceil((target - now.getTime()) / 86_400_000)
}

export function formatCurrency(value: number | null, currency: string | null): string {
  if (value == null) return '—'
  try {
    return new Intl.NumberFormat('es-CO', {
      style: currency ? 'currency' : 'decimal',
      currency: currency ?? undefined,
      maximumFractionDigits: 0,
    }).format(value)
  } catch {
    return `${value.toLocaleString('es-CO')}${currency ? ` ${currency}` : ''}`
  }
}
```

```ts
import { analyzeOpportunity } from '@/lib/agent/analyze'
import { generateWithOpenRouter } from '@/lib/agent/llm'

export const runtime = 'nodejs'
export const maxDuration = 60

export async function POST(req: Request) {
  try {
    const { text } = (await req.json()) as { text?: unknown }
    if (typeof text !== 'string' || text.trim().length === 0) {
      return Response.json({ error: 'Falta el texto de la convocatoria.' }, { status: 400 })
    }
    const analysis = await analyzeOpportunity(text, { generate: generateWithOpenRouter })
    return Response.json(analysis)
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Error desconocido al analizar.'
    return Response.json({ error: message }, { status: 500 })
  }
}
```

```ts
import type { OpportunityAnalysis } from '@/lib/agent/schema'

export async function analyzeClient(text: string): Promise<OpportunityAnalysis> {
  if (process.env.NEXT_PUBLIC_USE_FIXTURE === '1') {
    const { SAMPLE_ANALYSIS } = await import('./sample-analysis')
    await new Promise((resolve) => setTimeout(resolve, 600))
    return SAMPLE_ANALYSIS
  }

  const res = await fetch('/api/analyze', {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({ text }),
  })
  if (!res.ok) {
    const body = (await res.json().catch(() => ({}))) as { error?: string }
    throw new Error(body.error ?? `Error ${res.status} al analizar la convocatoria.`)
  }
  return res.json() as Promise<OpportunityAnalysis>
}
```

```tsx
'use client'

import { useState } from 'react'
import type { OpportunityAnalysis } from '@/lib/agent/schema'
import { analyzeClient } from '@/lib/ui/analyze-client'
import { OpportunityInput } from '@/components/opportunity-input'
import { AnalysisView } from '@/components/analysis/analysis-view'

      setStatus('error')

        value={text}
        onChange={setText}
        onAnalyze={run}
        collapsed={status === 'done'}
        loading={status === 'loading'}
        sourceName={analysis?.source.name}
      />

      {status === 'loading' && (
        <div className="flex flex-col gap-3">
          <Skeleton className="h-28 w-full" />
          <Skeleton className="h-40 w-full" />
        </div>

      {status === 'done' && analysis && <AnalysisView analysis={analysis} />}

}
```
